#TestBigEndian
include(TestBigEndian)
TEST_BIG_ENDIAN(PhysicaEndian)
if(${PhysicaEndian})
    set(PhysicaEndian PhysicaBigEndian)
else()
    set(PhysicaEndian PhysicaLittleEndian)
endif()
#CheckIntegerCast
PHYSICA_CHECK(CheckIntegerCast)
if(${RUN_RESULT_VAR} MATCHES 1)
    message(STATUS "Performing CheckIntegerCast --Failed")
endif()
message(STATUS "Performing CheckIntegerCast --Passed")
#CheckRightShift
PHYSICA_CHECK(CheckRightShift)
if(${RUN_RESULT_VAR} MATCHES 1)
    message(STATUS "Performing CheckRightShift --Failed")
endif()
message(STATUS "Performing CheckRightShift --Passed")
#CheckWordSize
PHYSICA_CHECK(CheckWordSize)
if(NOT ${RUN_RESULT_VAR} MATCHES 64 AND NOT ${RUN_RESULT_VAR} MATCHES 32)
    message(STATUS "Performing CheckWordSize --Failed")
    message(FATAL_ERROR "Unsupported word length.")
endif()
message(STATUS "Performing CheckWordSize --${RUN_RESULT_VAR}")
set(PhysicaWordSize ${RUN_RESULT_VAR})

message(STATUS)
configure_file(Config.h.in ${CMAKE_SOURCE_DIR}/include/Physica/Config.h)
